void update(BackupModel backup) {
  _db.execute('''
    UPDATE $table
    SET
      $columnTimestamp = ?,
      $columnSignature = ?
    WHERE $columnPath = ?;
    ''', [
    backup.timestamp!.millisecondsSinceEpoch,
    backup.signature,
    backup.path
  ]);
}
