static Future<Uint8List> signAsync(
    CryptoRSAPrivateKey key, Uint8List message) {
  Map<String, String> q = {};
  q['message'] = base64.encode(message);
  q['key'] = key.encode();
  return compute(
          (Map<String, String> q) => sign(
              CryptoRSAPrivateKey.decode(q['key']!),
              base64.decode(q['message']!)),
          q)
      .then((signature) => signature);
}
